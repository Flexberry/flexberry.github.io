    (
    • config
    • record
    )

    Inherited from ObjectListViewComponent: ember-flexberry/addon/components/object-list-view.js:538

    Hook for configurate rows.

    Parameters:

    • config Object
      Settings for row. See {{#crossLink "ObjectListView/defaultRowConfig:property"}}{{/crossLink}} property for details
    • record DS.Model
      The record in row

    Example:

    <!-- app/templates/employees.hbs -->
                    {{flexberry-objectlistview
                      ...
                      configurateRow=(action 'configurateRow')
                      ...
                    }}
                    
    // app/controllers/employees.js
                    import ListFormController from './list-form';
                    
                    export default ListFormController.extend({
                      actions: {
                        configurateRow(rowConfig, record) {
                          rowConfig.canBeDeleted = false;
                          if (record.get('isMyFavoriteRecord')) {
                            rowConfig.customClass += 'my-fav-record';
                          }
                        }
                      }
                    });
